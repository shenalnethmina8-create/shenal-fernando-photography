<script>
// LOADER
let count = 0;
let counter = document.getElementById("count");
let progress = document.getElementById("progress");
let loader = setInterval(() => {
    count++;
    counter.innerHTML = count + " / 100";
    progress.style.transform = `rotate(${count * 3.6}deg)`;
    if(count >= 100){
        clearInterval(loader);
        document.getElementById("loader").style.display = "none";
        document.getElementById("main").classList.remove("hidden");
    }
}, 25);

// CURSOR FOLLOW
const cursor = document.getElementById("cursor");
document.addEventListener("mousemove", e => {
    cursor.style.left = e.clientX + "px";
    cursor.style.top = e.clientY + "px";
});

// WELCOME CLICK → SMOKE → IMAGE PAGE
document.getElementById("welcomeBtn").addEventListener("click", () => {
    let smoke = document.getElementById("smoke");
    smoke.style.opacity = "0.9";
    setTimeout(() => {
        smoke.style.opacity = "0";
        document.getElementById("main").style.display = "none";
        document.getElementById("imagePage").style.display = "flex";
    }, 800);
});

// BOOKING FORM → WHATSAPP
document.getElementById("bookingForm").addEventListener('submit', function(e){
    e.preventDefault();
    let name = document.getElementById("clientName").value;
    let email = document.getElementById("clientEmail").value;
    let message = document.getElementById("clientMessage").value;
    let whatsappLink = `https://wa.me/94771345261?text=Name:${encodeURIComponent(name)}%0AEmail:${encodeURIComponent(email)}%0AMessage:${encodeURIComponent(message)}`;
    window.open(whatsappLink, "_blank");
});

// DATE DISPLAY
function updateDate(){
    let now = new Date();
    document.getElementById("dateCorner").innerText = now.toLocaleDateString();
}
updateDate();
setInterval(updateDate, 60000);
</script>
